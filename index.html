function createState(initial) {
  let value = initial;
  const subs = new Set();
  return {
    get: () => value,
    set: (v) => { value = v; subs.forEach(s => s(value)); },
    subscribe: (fn) => { subs.add(fn); return () => subs.delete(fn); }
  }
}
